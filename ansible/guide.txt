## Using ansible-container
## python3.5

# Create your parent folder
	$ mkdir hello-world
    $ cd hello-world

# Initialize ansible
	$ ansible-container init

# Create a sample app, in the sample its a flask app
# helloworld.py

# create requirements.txt

# create roles
	$ mkdir roles
	$ cd roles
	$ ansible-galaxy init flask

# define tasks/main.yml
# these are the processes inside the container: install, update or
# some other process you want to run inside the container

# define container.yml
# to provision the container

# build the service
	$ ansible-container build 

# run the container
	$ ansible-container run
